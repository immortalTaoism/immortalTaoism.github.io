一、安装JDK1.8

  • 先从orcle官网下载下最新的JDK “jdk-8u231-linux-x64.tar.gz”
  • 通过FTP工具的SSH连接方式，把文件传输到服务器上

先卸载默认的openjdk*

1、新建java目录

cd /usr/local
mkdir java

2、把”jdk-8u231-linux-x64.tar.gz”放到/usr/local/java

cd /usr/local/java
cp /root/jdk-8u231-linux-x64.tar.gz .

2、配置环境变量

vim /etc/profile

# 文件最底部添加
export JAVA_HOME=/usr/local/java/jdk1.8.0_231
export CLASSPATH=.:$JAVA_HOME/jre/lib/rt.jar:$JAVA_HOME/lib/dt.jar:$JAVA_HOME/lib/tools.jar
export PATH=$PATH:$JAVA_HOME/bin

3、查看是否安装成功

java -version

如果java -verison没生效，可以使用下面命令

source /etc/profile
java -version

一、安装MySQL

先从官网下载最新的社区版本的MySQL，因为centos是属于red hat下的,记得下载Red Hat Enterprise Linux / Oracle Linux，

  • 下载地址:https://dev.mysql.com/downloads/mysql/

mysql-8.0.18-1.el7.x86_64.rpm-bundle.tar

1、新建目录

cd /usr/local/
mkdir mysql

2、把“mysql-8.0.18-1.el7.x86_64.rpm-bundle.tar”放到/usr/local/mysql

cd /usr/local/mysql
cp /root/mysql-8.0.18-1.el7.x86_64.rpm-bundle.tar .

3、解压文件，解压完会得到很多.rpm的安装包

tar -xvf mysql-8.0.18-1.el7.x86_64.rpm-bundle.tar

4、安装

# linux都会自带一个默认的数据库，需要先卸载掉它
# 1.查看当前是否有mariadb
rpm -qa | grep mariadb
# 2.有则卸载它
rpm -e 上条命令查询出来的mariadb名称 --nodeps

# 正式安装mysql
rpm -ivh mysql-community-common-8.0.18-1.el7.x86_64.rpm --nodeps --force
rpm -ivh mysql-community-libs-8.0.18-1.el7.x86_64.rpm --nodeps --force
rpm -ivh mysql-community-client-8.0.18-1.el7.x86_64.rpm --nodeps --force
rpm -ivh mysql-community-server-8.0.18-1.el7.x86_64.rpm --nodeps --force

5、初始化mysql

mysqld --initialize

# 授权
chown mysql:mysql /var/lib/mysql -R;
# 启动mysql
systemctl start mysqld.service;
# 设置mysql为开启自启
systemctl enable mysqld;

6、设置防火墙规则

# 检查防火墙是否工作状态
systemctl status firewalld
# 没有防火墙没启动则启动一下
systemctl start firewalld;
# 设置3306端口
firewall-cmd --zone=public --add-port=3306/tcp --permanent
# 应用配置
firewall-cmd --reload

6、查看mysql默认生成的密码，需要从日志中找

cat /var/log/mysqld.log | grep password

执行完就会得到类似下面的信息,其中”QFCYu)rrH4.P”就是密码

cat /var/log/mysqld.log | grep password
QFCYu)rrH4.P

7、登录到mysql

mysql -uroot -p # 之后根据提示输入密码

# 修改密码为applejam
alter user 'root'@'localhost' identified by 'applejam';

# 设置mysql,root用户可以远程连接
use mysql;
update user set host = '%' where user = 'root';

# 更新设置
flush privileges;

用navicat连接的时候可能会遇到密码规则的问题，可以使用下面命令解决

mysql -uroot -p # 之后根据提示输入密码

use mysql;

alter user 'root'@'%' identified with mysql_native_password by 'applejam';

flush privileges;
